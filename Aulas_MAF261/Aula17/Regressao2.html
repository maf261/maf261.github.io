Exemplo 10.1

Verificar, utilizando os dados amostrais fornecidos abaixo, se a temperatura tem influ√™ncia significativa sobre o comprimento de uma barra de a√ßo. Utilize o modelo linear de \(1^{\circ}\) grau e o n√≠vel de \(5\%\) de signific√¢ncia.

Temperatura \((^{\circ}C)\) 10 15 20 25 30
Comprimento 1003 1005 1010 1011 1014 
### Dados do Exercicio: Temperatura x Comprimento
### x = Temperatura e y = Comprimento
############################################

# Entrada de Dados
x <- c(10,15,20,25,30)
y <- c(1003,1005,1010,1011,1014)

Calculando as express√µes sem usar fun√ß√µes do R:

plot(x,y, pch=19, col="red3") # Temperatura x Comprimento

(n <- length(x)) # tamanho da amostra
## [1] 5
(xbar <- mean(x)) # m√©dia das Temperaturas
## [1] 20
(ybar <- mean(y)) # m√©dia dos Comprimentos
## [1] 1008.6
(s2x <- var(x)) # vari√¢ncia das Temperaturas
## [1] 62.5
(s2y <- var(y)) # vari√¢ncia dos Comprimentos
## [1] 20.3
(sxy <- (sum(x*y)-n*xbar*ybar)/(n-1)) # covari√¢ncia entre Comprimento e Temperatura
## [1] 35
(rxy <- cor(x,y)) # correla√ß√£o entre Comprimento e Temperatura
## [1] 0.9826074
## Ajuste de m√≠nimos quadrados
## estimativas dos coeficientes
(beta1 <- sxy/s2x)
## [1] 0.56
(beta0 <- ybar-xbar*beta1)
## [1] 997.4
(c(beta0,beta1))
## [1] 997.40   0.56
## Coeficiente de determina√ß√£o
R2 <- (sxy^2)/(s2y*s2x)
R2
## [1] 0.9655172
(rxy)^2 # rela√ß√£o de R2 com rxy
## [1] 0.9655172
beta1*sxy/s2y # rela√ß√£o de R2 com beta1
## [1] 0.9655172
## Valores ajustados e res√≠duos
ych <- beta0 + beta1*x # valores ajustados
ei <- y - ych # res√≠duos
## plotando a reta ajustada
plot(x,y, pch=19, col="red3", main="Gr√°fico de pontos", xlab="Temperatura",
ylab="Comprimento")
lines(x[order(x)], ych[order(x)],col="blue3", lwd=2)

## An√°lise gr√°fica para verifica√ß√£o das suposi√ß√µes
## do modelo (normalidade e igualdade de vari√¢ncias)
# histograma
hist(ei, col="bisque", main="Histograma dos res√≠duos", xlab="Res√≠duos",
ylab="Frequ√™ncia", cex.main=1.25)

# normal qq-plot dos res√≠duos
qqnorm(ei)
qqline(ei)

# res√≠duos x valores ajustados
plot(ych,ei, main="Res√≠duos x Ajustados", pch=19, col="red3")

# res√≠duos x ordem (timeplot)
plot(ei, main="Res√≠duos x ordem", pch=19, col="red3")
lines(ei, lwd=2, col="blue3")
